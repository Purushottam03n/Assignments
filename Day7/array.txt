number n into an array and finally display the output.

index=0

for ((i=2; i<=num; i++))

	while [ $((num%$i)) == 0 ]

4. Write a Program to show Sum of three Integer adds to ZERO





declare -a arr

arr=("-1" "-3" "5" "6" "2" "4" "-8")

n=${#arr[*]}

for (( i=0; i<n; i++ ))

do
	for (( j=$i+1; j<n; j++ ))

	do
		for (( k=$j+1; k<n; k++ ))

		do
			sum=$(( ${arr[i]} + ${arr[j]} + ${arr[k]}))

			if [ $sum -eq 0 ]

			then
				echo "Three Intergers are: " ${arr[i]} ${arr[j]} ${arr[k]}

				exit 0
			fi
		done
	done
done
